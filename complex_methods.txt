
  # Compute eigenvalues and return a Tensor of Complex numbers.
  # This method is specifically added for control theory support in cryspace.
  def eigvals_c
    self.assert_square_matrix
    a = self.dup(Num::ColMajor)
    n = a.shape[0]
    
    # We use manual pointer buffers to avoid macro issues and handle complex parts
    wr = Pointer(T).malloc(n)
    wi = Pointer(T).malloc(n)
    
    jobvl = 'N'.ord.to_u8
    jobvr = 'N'.ord.to_u8
    lda = n
    ldvl = 1
    ldvr = 1
    vl_dummy = T.zero
    vr_dummy = T.zero
    
    # Workspace query
    lwork = -1
    work_query = T.zero
    info = 0
    
    {% if T == Float32 %}
      LibLapack.sgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, pointerof(vl_dummy), pointerof(ldvl), pointerof(vr_dummy), pointerof(ldvr), 
        pointerof(work_query), pointerof(lwork), pointerof(info))
      lwork = work_query.to_i32
      work = Pointer(Float32).malloc(lwork)
      LibLapack.sgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, pointerof(vl_dummy), pointerof(ldvl), pointerof(vr_dummy), pointerof(ldvr), 
        work, pointerof(lwork), pointerof(info))
    {% elsif T == Float64 %}
      LibLapack.dgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, pointerof(vl_dummy), pointerof(ldvl), pointerof(vr_dummy), pointerof(ldvr), 
        pointerof(work_query), pointerof(lwork), pointerof(info))
      lwork = work_query.to_i32
      work = Pointer(Float64).malloc(lwork)
      LibLapack.dgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, pointerof(vl_dummy), pointerof(ldvl), pointerof(vr_dummy), pointerof(ldvr), 
        work, pointerof(lwork), pointerof(info))
    {% else %}
      raise "eigvals_c not implemented for #{T}"
    {% end %}
    
    raise "LAPACK geev returned #{info}" if info != 0
    
    res = Tensor(Complex, CPU(Complex)).new([n])
    n.times do |i|
      res.to_unsafe[i] = Complex.new(wr[i].to_f64, wi[i].to_f64)
    end
    res
  end

  # Compute eigenvalues and eigenvectors with complex support.
  def eig_c
    self.assert_square_matrix
    a = self.dup(Num::ColMajor)
    n = a.shape[0]
    
    wr = Pointer(T).malloc(n)
    wi = Pointer(T).malloc(n)
    vl = Tensor(T, S).new([n, n])
    vr = Tensor(T, S).new([n, n])
    
    jobvl = 'V'.ord.to_u8
    jobvr = 'V'.ord.to_u8
    lda = n
    ldvl = n
    ldvr = n
    
    lwork = -1
    work_query = T.zero
    info = 0
    
    {% if T == Float32 %}
      LibLapack.sgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, vl.to_unsafe, pointerof(ldvl), vr.to_unsafe, pointerof(ldvr), 
        pointerof(work_query), pointerof(lwork), pointerof(info))
      lwork = work_query.to_i32
      work = Pointer(Float32).malloc(lwork)
      LibLapack.sgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, vl.to_unsafe, pointerof(ldvl), vr.to_unsafe, pointerof(ldvr), 
        work, pointerof(lwork), pointerof(info))
    {% elsif T == Float64 %}
      LibLapack.dgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, vl.to_unsafe, pointerof(ldvl), vr.to_unsafe, pointerof(ldvr), 
        pointerof(work_query), pointerof(lwork), pointerof(info))
      lwork = work_query.to_i32
      work = Pointer(Float64).malloc(lwork)
      LibLapack.dgeev(pointerof(jobvl), pointerof(jobvr), pointerof(n), a.to_unsafe, pointerof(lda),
        wr, wi, vl.to_unsafe, pointerof(ldvl), vr.to_unsafe, pointerof(ldvr), 
        work, pointerof(lwork), pointerof(info))
    {% else %}
      raise "eig_c not implemented for #{T}"
    {% end %}
    
    raise "LAPACK geev returned #{info}" if info != 0
    
    res_w = Tensor(Complex, CPU(Complex)).new([n])
    n.times do |i|
      res_w.to_unsafe[i] = Complex.new(wr[i].to_f64, wi[i].to_f64)
    end
    {res_w, vr}
  end
